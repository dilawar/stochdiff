21      public IconSwitcher(DruButton but) {
22          target = but;
23          nstate = 0;
24          states = new String[4];
25          srcs = new String[4];
26      }
27  
28  
29      public void addResponse(String stat, String src) {
30          states[nstate] = stat;
31          srcs[nstate] = src;
32          nstate += 1;
33      }
34  
35  
36      public void stateChanged(String s) {
37          for (int i = 0; i < nstate; i++) {
38              if (states[i].equals(s)) {
39                  target.setIconSource(srcs[i]);
40                  break;
41              }
42          }
43      }
