27      /**
28       * Aligns the first <code>rows</code> * <code>cols</code>
29       * components of <code>parent</code> in
30       * a grid. Each component is as big as the maximum
31       * preferred width and height of the components.
32       * The parent is made just big enough to fit them all.
33       *
34       * @param rows number of rows
35       * @param cols number of columns
36       * @param initialX x location to start the grid at
37       * @param initialY y location to start the grid at
38       * @param xPad x padding between cells
39       * @param yPad y padding between cells
40       */
41      public static void makeGrid(Container parent,
42                                  int rows, int cols,
43                                  int initialX, int initialY,
44                                  int xPad, int yPad) {
45          SpringLayout layout;
46          try {
47              layout = (SpringLayout)parent.getLayout();
48          } catch (ClassCastException exc) {
49              System.err.println("The first argument to makeGrid must use SpringLayout.");
50              return;
51          }
52  
53          Spring xPadSpring = Spring.constant(xPad);
54          Spring yPadSpring = Spring.constant(yPad);
55          Spring initialXSpring = Spring.constant(initialX);
56          Spring initialYSpring = Spring.constant(initialY);
57          int max = rows * cols;
58  
59          //Calculate Springs that are the max of the width/height so that all
60          //cells have the same size.
61          Spring maxWidthSpring = layout.getConstraints(parent.getComponent(0)).
62                                  getWidth();
63          Spring maxHeightSpring = layout.getConstraints(parent.getComponent(0)).
64                                   getWidth();
65          for (int i = 1; i < max; i++) {
66              SpringLayout.Constraints cons = layout.getConstraints(
67                                                  parent.getComponent(i));
68  
69              maxWidthSpring = Spring.max(maxWidthSpring, cons.getWidth());
70              maxHeightSpring = Spring.max(maxHeightSpring, cons.getHeight());
71          }
72  
73          //Apply the new width/height Spring. This forces all the
74          //components to have the same size.
75          for (int i = 0; i < max; i++) {
76              SpringLayout.Constraints cons = layout.getConstraints(
77                                                  parent.getComponent(i));
78  
79              cons.setWidth(maxWidthSpring);
80              cons.setHeight(maxHeightSpring);
81          }
82  
83          //Then adjust the x/y constraints of all the cells so that they
84          //are aligned in a grid.
85          SpringLayout.Constraints lastCons = null;
86          SpringLayout.Constraints lastRowCons = null;
87          for (int i = 0; i < max; i++) {
88              SpringLayout.Constraints cons = layout.getConstraints(
89                                                  parent.getComponent(i));
90              if (i % cols == 0) { //start of new row
91                  lastRowCons = lastCons;
92                  cons.setX(initialXSpring);
93              } else { //x position depends on previous component
94                  cons.setX(Spring.sum(lastCons.getConstraint(SpringLayout.EAST),
95                                       xPadSpring));
96              }
97  
98              if (i / cols == 0) { //first row
99                  cons.setY(initialYSpring);
100             } else { //y position depends on previous row
101                 cons.setY(Spring.sum(lastRowCons.getConstraint(SpringLayout.SOUTH),
102                                      yPadSpring));
103             }
104             lastCons = cons;
105         }
106 
107         //Set the parent's size.
108         SpringLayout.Constraints pCons = layout.getConstraints(parent);
109         pCons.setConstraint(SpringLayout.SOUTH,
110                             Spring.sum(
111                                 Spring.constant(yPad),
112                                 lastCons.getConstraint(SpringLayout.SOUTH)));
113         pCons.setConstraint(SpringLayout.EAST,
114                             Spring.sum(
115                                 Spring.constant(xPad),
116                                 lastCons.getConstraint(SpringLayout.EAST)));
117     }

119     /* Used by makeCompactGrid. */
120     private static SpringLayout.Constraints getConstraintsForCell(
121         int row, int col,
122         Container parent,
123         int cols) {
124         SpringLayout layout = (SpringLayout) parent.getLayout();
125         Component c = parent.getComponent(row * cols + col);
126         return layout.getConstraints(c);
127     }

129     /**
130      * Aligns the first <code>rows</code> * <code>cols</code>
131      * components of <code>parent</code> in
132      * a grid. Each component in a column is as wide as the maximum
133      * preferred width of the components in that column;
134      * height is similarly determined for each row.
135      * The parent is made just big enough to fit them all.
136      *
137      * @param rows number of rows
138      * @param cols number of columns
139      * @param initialX x location to start the grid at
140      * @param initialY y location to start the grid at
141      * @param xPad x padding between cells
142      * @param yPad y padding between cells
143      */
144     public static void makeCompactGrid(Container parent,
145                                        int rows, int cols,
146                                        int initialX, int initialY,
147                                        int xPad, int yPad) {
148         SpringLayout layout;
149         try {
150             layout = (SpringLayout)parent.getLayout();
151         } catch (ClassCastException exc) {
152             System.err.println("The first argument to makeCompactGrid must use SpringLayout.");
153             return;
154         }
155 
156         //Align all cells in each column and make them the same width.
157         Spring x = Spring.constant(initialX);
158         for (int c = 0; c < cols; c++) {
159             Spring width = Spring.constant(0);
160             for (int r = 0; r < rows; r++) {
161                 width = Spring.max(width,
162                                    getConstraintsForCell(r, c, parent, cols).
163                                    getWidth());
164             }
165             for (int r = 0; r < rows; r++) {
166                 SpringLayout.Constraints constraints =
167                     getConstraintsForCell(r, c, parent, cols);
168                 constraints.setX(x);
169                 constraints.setWidth(width);
170             }
171             x = Spring.sum(x, Spring.sum(width, Spring.constant(xPad)));
172         }
173 
174         //Align all cells in each row and make them the same height.
175         Spring y = Spring.constant(initialY);
176         for (int r = 0; r < rows; r++) {
177             Spring height = Spring.constant(0);
178             for (int c = 0; c < cols; c++) {
179                 height = Spring.max(height,
180                                     getConstraintsForCell(r, c, parent, cols).
181                                     getHeight());
182             }
183             for (int c = 0; c < cols; c++) {
184                 SpringLayout.Constraints constraints =
185                     getConstraintsForCell(r, c, parent, cols);
186                 constraints.setY(y);
187                 constraints.setHeight(height);
188             }
189             y = Spring.sum(y, Spring.sum(height, Spring.constant(yPad)));
190         }
191 
192         //Set the parent's size.
193         SpringLayout.Constraints pCons = layout.getConstraints(parent);
194         pCons.setConstraint(SpringLayout.SOUTH, y);
195         pCons.setConstraint(SpringLayout.EAST, x);
196     }
