55      public void paintText(Graphics2D g) {
56          clearRegions();
57  
58          g.setColor(Color.black);
59          int ioff = 26;
60          int joff = 20;
61          for (XType xt : typeStore.getTypes()) {
62  
63              g.setFont(FontStore.instance().getDefaultFont());
64              drawType(g, ioff, joff, xt, "type");
65  
66              int xr = ioff - 18;
67              if (xt.hasFields()) {
68                  if (xt.isExpanded()) {
69                      g.drawString(" - ", xr, joff);
70                      addRegion(xr, joff + 4, 18, 16, xt, "collapse", Region.CLICK);
71  
72                  } else {
73                      g.drawString(" + ", ioff - 18, joff);
74                      addRegion(xr, joff + 4, 18, 16, xt, "expand", Region.CLICK);
75                  }
76                  joff += 16;
77  
78                  if (xt.isExpanded()) {
79                      g.setFont(FontStore.instance().getDefaultItalicFont());
80                      for (XType subt : xt.getSubtypes()) {
81                          drawType(g, ioff + 16, joff, subt, "subtype");
82                          joff += 16;
83                      }
84                  }
85              } else {
86                  joff += 16;
87              }
88          }
89  
90          fullTextHeight = joff;
91  //      addRegion(2, getHeight()-2, getWidth()-4, getHeight()-4 - joff, null, "_global", Region.DROP);
92  
93      }

99      private void drawType(Graphics2D g, int ioff, int joff, XType xt, String tag) {
100         if (xt == selectedType) {
101             g.setColor(Color.yellow);
102             g.fillRect(0, joff - 12, canvas.getWidth(), 16);
103         }
104 
105         g.setColor(Color.black);
106 
107         g.drawString(xt.getID(), ioff, joff);
108         addRegion(ioff, joff + 4, 60, 16, xt, tag, Region.DRAG_OR_DROP);
109     }

115     public void regionClicked(Region reg) {
116         String tag  = reg.getTag();
117 
118         if (tag != null) {
119             if (tag.equals("expand")) {
120                 ((XType)reg.getRef()).expand();
121 
122             } else if (tag.equals("collapse")) {
123                 ((XType)reg.getRef()).collapse();
124 
125             } else if (tag.equals("type")) {
126                 selectedType = (XType)reg.getRef();
127 
128                 notifyChange(selectedType);
129 
130             } else {
131                 E.error("doint know " + tag);
132             }
133         }

151     public void dropOn(Object src, Region reg) {
152         if ("_global".equals(reg.getTag())) {
153             dropGeneral(src);
154 
155         } else {
156 
157             Object tgt = reg.getRef();
158 
159             if (src instanceof WordBlock && tgt instanceof XType) {
160 
161                 WordBlock tbl = (WordBlock)src;
162                 XType xt = (XType)tgt;
163 
164                 pageDataStore.addItem(xt, tbl.getExtendedText());
165                 while (tbl != null) {
166                     tbl.setType(xt);
167                     tbl = tbl.getLinkee();
168                 }
169 
170             } else {
171                 E.error("wrong type  -ignoring drop of " + src + " on " + tgt);
172             }
173             repaint();
174         }
175

182     public void dropGeneral(Object src) {
183         if (src instanceof TextDisplayed) {
184             String txt = ((TextDisplayed)src).getDisplayText();
185             XType xt = new XType(txt);
186             typeStore.addIfNew(xt);
187 
188             if (src instanceof WordBlock) {
189                 ((WordBlock)src).weakSetType(typeStore.getTypeType());
190             }
191 
192         } else {
193             E.error("can only accept text displayed, not " + src);
194         }
195         repaint();
196     }
