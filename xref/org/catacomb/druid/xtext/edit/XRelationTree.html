76      private void drawRelation(Graphics2D g, int joff, XRelationType xr) {
77          int cw = canvas.getWidth();
78          if (xr == selectedType) {
79              g.setColor(Color.yellow);
80              g.fillRect(0, joff - 12, canvas.getWidth(), 16);
81          }
82  
83          g.setColor(Color.black);
84  
85          int txtoff = 80;
86  
87          g.drawString(xr.getID(), txtoff, joff);
88  
89          int sw = g.getFontMetrics().stringWidth(xr.getID());
90  
91          addRegion(txtoff, joff+4, sw, 16, xr, "main", Region.DRAG_OR_DROP);
92  
93          addRegion(5, joff + 4, txtoff-5, 16, xr, "left", Region.DROP);
94  
95          int w = cw - (txtoff - sw - 10);
96          if (w > 100) {
97              w = 100;
98          }
99          addRegion(txtoff+sw+5, joff + 4, w, 16, xr, "right", Region.DROP);
100     }

105     public void dropOn(Object src, Region reg) {
106         Object tgt = reg.getRef();
107         String tag = reg.getTag();
108         String txt = "err";
109         if (src instanceof TextDisplayed) {
110             txt = ((TextDisplayed)src).getDisplayText();
111         } else {
112             E.error("wrong typ drop " + src);
113         }
114 
115         if (tgt instanceof XRelationType) {
116             XRelationType xrt = (XRelationType)tgt;
117             XRelation xr = new XRelation(xrt);
118 
119             if (tag.equals("left")) {
120                 xr.setA(txt, src);
121             } else if (tag.equals("right")) {
122                 xr.setB(txt, src);
123             }
124             pageDataStore.addRelation(xr);
125 
126             if (src instanceof WordBlock) {
127                 ((WordBlock)src).weakSetType(typeStore.getThingType());
128 
129             }
130         }
131         repaint();
132     }
