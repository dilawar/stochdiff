54      private double[] fromScaledPoints(double[] va, double c, double scl) {
55          int n = va.length;
56          double[] ret = new double[n];
57          for (int i = 0; i < n; i++) {
58              ret[i] = (va[i] - c) / scl;
59          }
60          return ret;
61      }
