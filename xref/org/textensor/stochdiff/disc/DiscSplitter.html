15   * Take a structure expressed as TreePoints each of which knows their
16   * neighbors, and split it into radially and tangentially separated elements. The structure should already
17   * have been segment-sliced so that here we just subdivide the segments

36      public DiscSplitter(TreePoint[] pts, double d, HashMap<String, Double> resHM, double[] sl, double mar) {
37          srcPoints = pts;
38          resolution = new Resolution(d, resHM);
39          if (sl != null && sl.length > 0) {

50      public VolumeGrid buildGrid() {
51  
52          TreePoint firstpt = null;
53          // put them all in a set - take them out when they've been done;
54          wkpHS = new HashSet<TreePoint>();
55          for (TreePoint tp : srcPoints) {
56              wkpHS.add(tp);
57              // need one that has only one neighbor as the start point;
58              if (firstpt == null && tp.isEndPoint()) {
59                  firstpt = tp;
60              }
61          }
62          TreeUtil.parentizeFrom(firstpt, srcPoints);
63  
64          gridAL = new ArrayList<CurvedVolumeSlice>();
65          CurvedVolumeSlice vg0 = null;
66          wkpHS.remove(firstpt);
67  
68          recAdd(vg0, firstpt);
69  
70          VolumeGrid vgr = new VolumeGrid();
71  
72          vgr.importSmoothSlices(gridAL);
73          return vgr;
74  
75      }

81      private void recAdd(CurvedVolumeSlice pGrid, TreePoint tp) {
82          String lbl = tp.getLabel();
83  
84          tp.partBranchOffset = 0.;
85  
86  
87          for (TreePoint tpn : tp.getNeighbors()) {
88              if (wkpHS.contains(tpn)) {
89                  wkpHS.remove(tpn);
90  
91                  // if a terminal has a label, and the current point doesn't, then use it
92                  if (lbl == null && tpn.nnbr == 1 && tpn.getLabel() != null) {
93                      lbl = tpn.getLabel();
94                  }
95  
96                  CurvedVolumeSlice vg = null;
97                  if (tpn.subAreaPeer == tp) {
98                      // nothing to do for now - put line in when we
99                      // do the first child of tpn
100                     // E.info("skipping pt with peer " + tpn);
101 
102                 } else if (tp.subAreaPeer != null && tp.subAreaPeer == tp.parent) {
103                     // E.info("first pt after branch " + tpn);
104                     TreePoint par = tp.parent;
105                     E.info("starting a sub-branch at " + tp + " - " + tpn + " " + pGrid);
106 
107                     vg = baseGrid(tp, tpn, lbl);
108                     pGrid.subPlaneConnect(tp, tpn, vg, par.partBranchOffset);
109                     par.partBranchOffset += 2 * tpn.r;
110 
111                 } else {
112                     // normal case: make a new one or add a slice and connect
113                     // it up with the centres aligned
114                     if (pGrid == null) {
115                         vg = baseGrid(tp, tpn, lbl);
116 
117                     } else {
118                         // TODO - probably not what we want
119                         // too much mumerical diffusion if boxes can have gradually changing
120                         // sizes? restrict to a few dicrete multiples?
121                         vg = baseGrid(tp, tpn, lbl);
122                         pGrid.planeConnect(vg);
123                     }
124                 }
125 
126                 lbl = null; // only use it once
127                 if (vg != null) {
128                     gridAL.add(vg);
129                     recAdd(vg, tpn);
130                 } else {
131                     // skipped the point that is the start of a new segment
132                     // of different radius
133                     recAdd(pGrid, tpn);
134                 }
135             }
136         }
137     }

142     public CurvedVolumeSlice baseGrid(TreePoint tpa, TreePoint tpb, String lbl) {
143 
144         String rgn = tpa.regionClassWith(tpb);
145         double delta = resolution.getLocalDelta(tpa, tpb);
146 
147         CurvedVolumeSlice ret = new CurvedVolumeSlice(delta, tpa.getRadius(), tpb.getRadius());
148         ret.discFill(tpa, tpb, lbl, rgn, surfaceLayers, maxAR);
149 
150         return ret;
151     }
