45      public void discFill(Position pa, Position pb, String pointLabel, String regionLabel,
46                           double[] surfaceLayers, double maxAR) {
47          maxAspectRatio = maxAR;
48  
49          double axlen = Geom.distanceBetween(pa, pb);
50          Translation trans = Geom.translation(Geom.midpoint(pa, pb));
51          Vector vab = Geom.fromToVector(pa, pb);
52          double arotx = Geom.zElevation(vab);
53          double arotz = Geom.yzRotationAngle(vab);
54          GRotation rotx = Geom.aboutXRotation(arotx);
55          GRotation rotz = Geom.aboutZRotation(arotz);
56          Rotation rot = rotz.times(rotx);
57  
58          Vector vy = Geom.unitY();
59          Position prot = rot.getRotatedPosition(Geom.endPosition(vy));
60  
61  
62          // Position prwk = rotx.getRotatedPosition(Geom.endPosition(vy));
63          // Position prot = rotz.getRotatedPosition(prwk);
64  
65          double da = Geom.angleBetween(vab, Geom.getToVector(prot));
66  
67  
68          if (Math.abs(da) > 1.e-6) {
69              throw new RuntimeException("rotation angle miscalculation: residual angle is " + da);
70          }
71  
72  
73          elements = new ArrayList<CurvedVolumeElement>();
74  
75  
76          // center of the box at 0,0
77  
78  
79          // this is a little confusing. X and Y axes are used within the slice, but when these are
80          // turned into boxes, the slab of boxes is initially created in the X-Z plane before being rotated
81          // into place
82  
83          double maxr = Math.max(radiusa, radiusb);
84          double[] bdm = getRadialSplit(maxr, surfaceLayers);
85  
86          bdsa = getRadialSplit(radiusa, bdm.length, surfaceLayers);
87          bdsb = getRadialSplit(radiusb, bdm.length, surfaceLayers);
88  
89  
90          nazim = getAzimuthalSplits(maxr, bdm);
91  
92          radHM = new HashMap<Integer, ArrayList<CurvedVolumeElement>>();
93  
94          for (int ir = 0; ir < bdsa.length; ir++) {
95  
96  
97              double ra1 = (ir > 0 ? bdsa[ir-1] : 0);
98              double ra2 = bdsa[ir];
99  
100             double rb1 = (ir > 0 ? bdsb[ir-1] : 0);
101             double rb2 = bdsb[ir];
102 
103             double rc = ((ra1 + ra2) / 2 + (rb1 + rb2) / 2) / 2;
104 
105 
106             int na = nazim[ir];
107             double eltangle = 2. * Math.PI / na;
108 
109             double volouter = axlen * (ra2 * ra2 + rb2 * rb2 + ra2 * rb2) / 3.;
110             double volinner = axlen * (ra1 * ra1 + rb1 * rb1 + ra1 * rb1) / 3.;
111             double eltvol = (volouter - volinner) / na;
112 
113             double surfouter = 2 * Math.PI * (ra2 + rb2) / 2.;
114 
115             double carea = axlen * ((ra2 - ra1) + (rb2 - rb1)) / 2;
116 
117             double subarea = (axlen * 2 * Math.PI * (ra1 + rb1) / 2) / na;
118 
119 
120 
121             ArrayList<CurvedVolumeElement> azb = null;
122             double eltangleb = 1.;
123             if (ir > 0) {
124                 azb = radHM.get(ir-1);
125                 eltangleb = 2 * Math.PI / azb.size();
126             }
127 
128 
129             ArrayList<CurvedVolumeElement> az = new ArrayList<CurvedVolumeElement>();
130             for (int ia = 0; ia < na; ia++) {
131 
132                 double theta = ia * eltangle;
133 
134                 CurvedVolumeElement ve = new CurvedVolumeElement();
135                 if (regionLabel != null) {
136                     ve.setRegion(regionLabel);
137                 }
138 
139                 ve.setPositionIndexes(ir, ia);
140                 ve.setVolume(eltvol);
141 
142                 if (na > 1) {
143                     if (ia > 0) {
144                         az.get(ia-1).coupleTo(ve, carea);
145                     }
146                     if (ia == na - 1) {
147                         ve.coupleTo(az.get(0), carea);
148                     }
149                 }
150 
151                 if (ir > 1) {
152                     double thc = theta + 0.5 * eltangle;
153                     int ib = (int)(thc / eltangleb);
154                     azb.get(ib).coupleTo(ve, subarea);
155                 }
156 
157 
158 
159                 double thetaC = theta + 0.5 * eltangle;
160                 double vcx = rc * Math.cos(thetaC);
161                 double vcy = 0.;
162                 double vcz = rc * Math.sin(thetaC);
163                 Position cp = Geom.position(vcx, vcy, vcz);
164                 Position pr = rot.getRotatedPosition(cp);
165                 Position pc = trans.getTranslated(pr);
166                 ve.setCenterPosition(pc.getX(), pc.getY(), pc.getZ());
167 
168 
169                 double rca = (ra1 + ra2) / 2;
170                 double rcb = (rb1 + rb2) / 2;
171 
172                 double dtheta = eltangle / 2;
173                 if (dtheta > 1) {
174                     dtheta = 1;
175                 }
176                 double ha = 0.5 * axlen;
177 
178                 Position[] pbdry = new Position[4];
179                 pbdry[0] = Geom.position(rca * Math.cos(thetaC - dtheta), -ha, rca * Math.sin(thetaC - dtheta));
180                 pbdry[1] = Geom.position(rcb * Math.cos(thetaC - dtheta), ha, rcb * Math.sin(thetaC - dtheta));
181                 pbdry[2] = Geom.position(rcb * Math.cos(thetaC + dtheta), ha, rcb * Math.sin(thetaC + dtheta));
182                 pbdry[3] = Geom.position(rca * Math.cos(thetaC + dtheta), -ha, rca * Math.sin(thetaC + dtheta));
183 
184                 for (int ib = 0; ib < pbdry.length; ib++) {
185                     pbdry[ib] = trans.getTranslated(rot.getRotatedPosition(pbdry[ib]));
186                 }
187                 ve.setBoundary(pbdry);
188 
189 
190 
191 
192 
193                 TrianglesSet ts = makeTriangles(axlen, ra1, ra2, rb1, rb2, theta, eltangle);
194 
195                 ts.rotate(rot);
196                 ts.translate(trans);
197 
198                 ve.setTriangles(ts.getStripLengths(), ts.getPositions(), ts.getNormals());
199 
200                 if (ir == bdsa.length - 1) {
201                     ve.setSubmembrane();
202                     ve.setExposedArea(surfouter / na);
203 
204 
205 
206                     Position[] praw = new Position[4];
207                     praw[0] = Geom.position(ra2 * Math.cos(thetaC - dtheta), -ha, ra2 * Math.sin(thetaC - dtheta));
208                     praw[1] = Geom.position(rb2 * Math.cos(thetaC - dtheta), ha, rb2 * Math.sin(thetaC - dtheta));
209                     praw[2] = Geom.position(rb2 * Math.cos(thetaC + dtheta), ha, rb2 * Math.sin(thetaC + dtheta));
210                     praw[3] = Geom.position(ra2 * Math.cos(thetaC + dtheta), -ha, ra2 * Math.sin(thetaC + dtheta));
211                     Position[] ps = new Position[praw.length];
212                     for (int i = 0; i < praw.length; i++) {
213                         ps[i] = trans.getTranslated(rot.getRotatedPosition(praw[i]));
214                     }
215 
216                     ve.setSurfaceBoundary(ps);
217                 }
218 
219                 az.add(ve);
220                 elements.add(ve);
221             }
222             radHM.put(ir, az);
223         }
224     }

229     private TrianglesSet makeTriangles(double axlen, double ra1, double ra2, double rb1, double rb2, double theta,
230                                        double eltangle) {
231 
232         // initial layout: elements are in the x-z plane, bottom surface at y = -0.5 * axlen, top at y = 0.5 * axlen
233         // rotations measured up from the x axis,
234 
235         TrianglesSet ret = new TrianglesSet();
236 
237         double dth = Math.PI * 2. / 36.;
238         int npart = (int)(Math.round(eltangle / dth));
239         if (npart < 1) {
240             npart = 1;
241         }
242 
243         if (eltangle < 1.9 * Math.PI) {
244             TriangleStrip tss = makeEnd(axlen, ra1, ra2, rb1, rb2, theta, -1);
245             ret.add(tss);
246             TriangleStrip tst = makeEnd(axlen, ra1, ra2, rb1, rb2, theta + eltangle, 1);
247             ret.add(tst);
248         }
249 
250         if (ra1 > 1.e-7) {
251             TriangleStrip tsin = makeConeSurfacePart(axlen, ra1, rb1, theta, eltangle, -1, npart);
252             ret.add(tsin);
253         }
254         TriangleStrip tsout = makeConeSurfacePart(axlen, ra2, rb2, theta, eltangle, 1, npart);
255         ret.add(tsout);
256 
257 
258         TriangleStrip tsp = makeSliceSurface(-0.5 * axlen, ra1, ra2, theta, eltangle, -1, npart);
259         ret.add(tsp);
260 
261         TriangleStrip tsq = makeSliceSurface(0.5 * axlen, rb1, rb2, theta, eltangle, 1, npart);
262         ret.add(tsq);
263 
264         return ret;
265     }

271     private TriangleStrip makeSliceSurface(double dy, double r1, double r2, double theta, double eltangle,
272                                            int idir, int npart) {
273         TriangleStrip ret = new TriangleStrip();
274 
275         double xn = 0.;
276         double yn = idir;
277         double zn = 0;
278 
279         for (int i = 0; i <= npart; i++) {
280             double a = theta + i * eltangle / npart;
281             double ca = Math.cos(a);
282             double sa = Math.sin(a);
283             ret.addPoint(r1 * ca, dy, r1 * sa, xn, yn, zn);
284             ret.addPoint(r2 * ca, dy, r2 * sa, xn, yn, zn);
285 
286         }
287         if (idir > 0) {
288             ret.flip();
289         }
290         return ret;
291     }

295     private TriangleStrip makeConeSurfacePart(double axlen, double ra, double rb,
296             double theta, double eltangle, int idir, int npart) {
297         TriangleStrip ret = new TriangleStrip();
298 
299 
300 
301         double ay = Math.atan2(rb - ra, axlen);
302         double fy = Math.sin(ay);
303         double fr = Math.cos(ay);
304 
305 
306         double am = -0.5 * axlen;
307         double ap = 0.5 * axlen;
308 
309         for (int i = 0; i < npart + 1; i++) {
310             double a = theta + i * eltangle / npart;
311 
312             double ca = Math.cos(a);
313             double sa = Math.sin(a);
314             double xn = fr * idir * ca;
315             double yn = -fy * idir;  // TODO check sign
316             double zn = fr * idir * sa;
317 
318             ret.addPoint(ra * ca, am, ra* sa,  xn, yn, zn);
319             ret.addPoint(rb * ca, ap, rb* sa,  xn, yn, zn);
320         }
321         if (idir < 0) {
322             ret.flip();
323         }
324 
325         return ret;
326     }

330     private TriangleStrip makeEnd(double axlen, double ra1, double ra2, double rb1, double rb2, double theta, int idir) {
331         TriangleStrip ret = new TriangleStrip();
332 
333         double y = -0.5 * axlen;
334         double ct = Math.cos(theta);
335         double st = Math.sin(theta);
336 
337         double xn = idir * st;
338         double yn = 0.;
339         double zn = -idir * ct;
340 
341         ret.addPoint(ra1 * ct, y, ra1 * st, xn, yn, zn);
342         ret.addPoint(ra2 * ct, y, ra2 * st, xn, yn, zn);
343         y = 0.5 * axlen;
344         ret.addPoint(rb1 * ct, y, rb1 * st, xn, yn, zn);
345         ret.addPoint(rb2 * ct, y, rb2 * st, zn, yn, zn);
346 
347         if (idir < 0) {
348             ret.flip();
349         }
350 
351         return ret;
352     }

357     private int[] getAzimuthalSplits(double radius, double[] bdm) {
358         int[] ret = new int[bdm.length];
359         int npre = 1;
360         for (int i = 0; i < bdm.length; i++) {
361             double rin = (i > 0 ? bdm[i-1] : 0);
362             double rout = bdm[i];
363             double rc = (rin + rout) / 2;
364 
365             double dr = rout - rin;
366             double circ = 2 * Math.PI * rc;
367 
368             int nfac = (int)Math.round(Math.ceil((circ / (dr * maxAspectRatio)) / npre));
369             ret[i] = npre * nfac;
370             npre = ret[i];
371         }
372         return ret;
373     }

383     private double[] getRadialSplit(double r, int ansplit, double[] sla) {
384         double[] ret = null;
385         int nsplit = ansplit;
386 
387         double rr = r;
388         int nsur = 0;
389         while (nsur < sla.length && rr > 2 * sla[nsur]) {
390             rr -= sla[nsur];
391             nsur += 1;
392         }
393         int nre = 1;
394         if (ansplit > 0) {
395             nre = ansplit - nsur;
396         } else {
397             nre = (int)Math.round(rr / baseDelta);
398             if (nre < 1) {
399                 nre = 1;
400             }
401         }
402 
403 
404         ret = new double[nre + nsur];
405         for (int i = 0; i < nre; i++) {
406             ret[i] = ((i + 1.)/(nre)) * rr;
407         }
408         for (int i = 0; i < nsur; i++) {
409             ret[nre + i] = ret[nre + i - 1] + sla[nsur - 1 - i];
410         }
411 
412         if (Math.abs(ret[nre + nsur - 1] - r) > 1.e-6) {
413             E.error("radius miscount " + r + " " + ret[nre + nsur - 1]);
414         }
415         String s = " ";
416         for (int i = 0; i < ret.length; i++) {
417             s += ret[i] + " ";
418         }
419         return ret;
420     }

432     public void planeConnect(CurvedVolumeSlice vg) {
433         double[] ras = getRadii(1);
434         int[] zas = getNazimuthals();
435 
436         double[] rbs = vg.getRadii(0);
437         int[] zbs = vg.getNazimuthals();
438 
439         double eps = 1.e-6;
440 
441         for (int ira = 0; ira < ras.length; ira++) {
442             double ra = ras[ira];
443             double ra0 = (ira > 0 ? ras[ira-1] : 0);
444 
445             for (int irb = 0; irb < rbs.length; irb++) {
446                 double rb = rbs[irb];
447                 double rb0 = (irb > 0 ? rbs[irb-1] : 0);
448 
449                 if (rb < ra0 + eps) {
450                     // b elt completely below a elt
451                 } else if (rb0 > ra - eps) {
452                     // b elt comletely above a
453 
454                 } else {
455                     // they overlap
456                     //   E.info("olrings " + ira + " " + irb + "     " + ra0 + " " + ra + "    " + rb0 + " " + rb);
457 
458 
459                     double ro0 = (ra0 > rb0 ? ra0 : rb0);
460                     double ro1 = (ra < rb ? ra : rb);
461                     double olarea = Math.PI * (ro0 * ro0  +  ro1 * ro1  +  ro0 * ro1) / 3.;
462 
463                     int na = zas[ira];
464                     int nb = zbs[irb];
465 
466                     if (na == nb) {
467                         // E.info("PCexact " + ira + " " + irb + " " + na);
468                         double carea = olarea / na;
469                         // they match up exactly - simple
470                         for (int iz = 0; iz < na; iz++) {
471                             getRAElement(ira, iz).coupleTo(vg.getRAElement(irb, iz), carea);
472 
473                         }
474 
475                     } else {
476                         double da = 1. / na;
477                         double db = 1. / nb;
478 
479                         int izb = 0;
480 
481                         for (int iza = 0; iza < na; iza++) {
482                             double a0 = iza * da;
483                             double a1 = a0 + da;
484 
485                             while (izb * db < a1 - eps) {
486                                 double b0 = izb * db;
487                                 double b1 = b0 + db;
488                                 double fc = Math.min(b1, a1) - Math.max(a0, b0);
489 
490                                 if (fc > eps) {
491                                     getRAElement(ira, iza).coupleTo(vg.getRAElement(irb, izb), fc * olarea);
492                                 }
493                                 // E.info("PCol " + ira + " " + irb + " " + na + " " + nb + "    " + iza + " " + izb);
494 
495                                 b0 = b1;
496                                 b1 += db;
497                                 izb += 1;
498                             }
499                             izb -= 1;
500 
501                         }
502 
503                     }
504                 }
505             }
506 
507 
508         }
509 
510         // vg is the next slice, startiong at our pb
511     }
