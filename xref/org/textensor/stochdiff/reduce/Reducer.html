58      public void reduce() {
59          E.info("Starting reduce");
60          icon = sdr.initialConditions;
61          afv = icon.getFloatValuedElements();
62          nv = afv.size();
63  
64          // these don't actually matter as it is all linear
65          // all concentrations are set to x0 and then they are separately varied by dx to
66          // compute the matrix of derivatives of the concentrations with respect to the accessible variables
67          double x0 = 10.0;
68          double dx = 0.1;
69  
70          E.info("The template provides " + nv + " accessible concentration values");
71  
72          // StaticCalc maps the model onto the array of concentrations for each element
73          staticCalc = new StaticCalc(sdr);
74          staticCalc.init();
75          vols = staticCalc.getVolumes();
76          nspec = staticCalc.getNSpec();
77          nel = staticCalc.getNel();
78  
79  
80          // the target concentrations from the SDState object
81          ctgt = sds.getConc2();
82  
83          // c0 is the array of unperturbed concentrations for all elements and species
84          for (int j = 0; j < nv; j++) {
85              afv.get(j).setValue(x0);
86          }
87          double[] c0 = staticCalc.getConcentrations();
88  
89  
90          int nconc = c0.length;
91          E.info("The template provides " + nv + " variable quantities");
92          E.info("After discretization there are " + nconc + " state variables arising from " + nspec + " species in "
93                 + nel + " elements");
94  
95          // bm will contain the derivatives of the concentrations with respect to the variables
96          double[][] bm = new double[nconc][nv];
97  
98          for (int i = 0; i < nv; i++) {
99              // set all initial values to x0 except for variable i which is set to x0 + dx
100             for (int j = 0; j < nv; j++) {
101                 afv.get(j).setValue(x0);
102             }
103             afv.get(i).setValue(x0 + dx);
104             double[] cwk = staticCalc.getConcentrations();
105 
106             // numerical derivatives go in bm
107             for (int k = 0; k < nconc; k++) {
108                 bm[k][i] = (cwk[k] - c0[k]) / dx;
109             }
110         }
111 
112 
113 
114         // if N variables give rise to M measures, (N << M) then we have
115         // mes(1,...M) = b * var(1,...,N);
116         // and
117         // bT mes = bT * b * var
118         // where bT is the transpose of b
119         // bT * b is square and invertible so
120         // var = inverse(bT * b) * bT * mes
121 
122         // tgtconc is like ctgt, but flattened into a single array
123         double[] tgtconc = sds.getConc1();
124 
125 
126         // evaluate var = inverst(bT * b) * bT * mes
127         Matrix b = new Matrix(bm);
128         Matrix bt = b.transpose();
129         Matrix mtgtconc = new Matrix(Matrix.COLUMN, tgtconc);
130         Matrix btb = bt.times(b);
131         Matrix btbi = btb.inverse();
132         Matrix res = btbi.times(bt.times(mtgtconc));
133 
134         // res is a matrix. newvars is just a plain array of the contents
135         double[] fit = res.getColumn(0);
136 
137 
138         // if there are any constraints in the initial conditions file, then modify the fit
139         // accordingly
140         String[] spres = icon.getTotalPreserved();
141         if (spres.length > 0) {
142 
143             // number of constraints
144             int ncn = spres.length;
145 
146             int[] ispeccon = staticCalc.getSpecieIndexes(spres);
147             String ss = "";
148             for (int i = 0; i < ncn; i++) {
149                 ss += " " + spres[i] + "(" + ispeccon[i] + ") ";
150             }
151             E.info(" " + ncn + " constraint(s) on totals for species " + ss);
152 
153             // q is the constraint matrix
154             double[][] q = new double[ncn][nv];
155 
156             // qtgt contains the values for each constraint (desired result of q * fit)
157             double[] qtgt = new double[ncn];
158 
159             for (int icn = 0; icn < ncn; icn++) {
160                 int ispec = ispeccon[icn];
161                 for (int iel = 0; iel < nel; iel++) {
162                     for (int i = 0; i < nv; i++) {
163                         q[icn][i] += vols[iel] * bm[iel * nspec + ispec][i];
164                     }
165                     qtgt[icn] += vols[iel] * ctgt[iel][ispec];
166                 }
167             }
168 
169             // The matrix calculation is similar in spirit to the above expression for the fit,
170             // but a little more complicated. c is the target value of the constraints (qtgt above)
171             // fit_constrained = fit - (BtB)inv Q (Qt (BtB)inv Q)inv (Qt fit - c)
172             Matrix mq = new Matrix(q);
173             Matrix mr = new Matrix(Matrix.COLUMN, fit);
174             Matrix mc = new Matrix(Matrix.COLUMN, qtgt);
175 
176             Matrix wk1 = mq.times(mr).subtract(mc);
177             Matrix wk2 = mq.times(btbi).times(mq.transpose());
178             Matrix wk3 = mq.transpose().times(wk2.inverse());
179             Matrix wk4 = btbi.times(wk3).times(wk1);
180             Matrix mrcon = mr.subtract(wk4);
181 
182             fit = mrcon.getColumnData();
183         }
184 
185 
186         // display the resulting fit.
187         // The second argument contains the concentrations that come from b * fit instead of
188         // via the StaticCalc element. These should be the same except where there are
189         // nonlinearities in the forward process as occur if some of the initial concentrations are
190         // negative.
191         showFit(fit, b.times(new Matrix(Matrix.COLUMN, fit)).getColumnData());
192 
193 
194 
195         for (int j = 0; j < nv; j++) {
196             afv.get(j).setValue(fit[j]);
197         }
198         String stxt = icon.xmlSerialize();
199         File fout = new File("reduce-fit.xml");
200         FileUtil.writeStringToFile(stxt, fout);
201         E.info("New initial conditions have been written to " + fout);
202         E.info("Best fit xml is:\n " + stxt);
203     }

210     public void showFit(double[] newvars, double[] mcdat) {
211         for (int j = 0; j < nv; j++) {
212             afv.get(j).setValue(newvars[j]);
213         }
214 
215         double[][] cfit = staticCalc.getElementConcentrations();
216 
217         E.info("Average concentrations by species: ");
218         for (int ispec = 0; ispec < nspec; ispec++) {
219             double a = 0;
220             double b = 0;
221             double c = 0;
222             double vtot = 0;
223             for (int iel = 0; iel < nel; iel++) {
224                 a += vols[iel] * ctgt[iel][ispec];
225                 b += vols[iel] * cfit[iel][ispec];
226                 c += vols[iel] * mcdat[iel * nspec + ispec];
227                 vtot += vols[iel];
228             }
229             a /= vtot;
230             b /= vtot;
231             E.info("    species " + ispec + " target=" + a + "  fit=" + b);
232 
233         }
