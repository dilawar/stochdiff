17      public static SDRun read(File modelFile) {
18  
19          ReflectionInstantiator rin = new ReflectionInstantiator();
20          rin.checkAddPackage(new ModelBase());
21          rin.checkAddPackage(new NeuroMLBase());
22  
23          XMLReader xmlr = new XMLReader(rin);
24  
25  
26          SDRun sdm = (SDRun)readFile(modelFile, xmlr);
27  
28          File parentFile = modelFile.getParentFile();
29          File freact = getFile(parentFile, sdm.reactionSchemeFile);
30          File fmorph = getFile(parentFile, sdm.morphologyFile);
31          File fstim = getFile(parentFile, sdm.stimulationFile);
32          File finitc = getFile(parentFile, sdm.initialConditionsFile);
33          //<--RO
34          //------------------
35          File foutput = getFile(parentFile, sdm.outputSchemeFile);
36          //------------------
37          //RO-->
38  
39  
40          sdm.setReactionScheme((ReactionScheme)readFile(freact, xmlr));
41          Object ob = readFile(fmorph, xmlr);
42          if (ob instanceof Transitional) {
43              ob = ((Transitional)ob).getFinal();
44          }
45          sdm.setMorphology((Morphology)ob);
46          sdm.setStimulationSet((StimulationSet)readFile(fstim, xmlr));
47          sdm.setInitialConditions((InitialConditions)readFile(finitc, xmlr));
48          //<--RO
49          //------------------
50          sdm.setOutputScheme((OutputScheme)readFile(foutput, xmlr));
51          //------------------
52          //RO-->
53          return sdm;
54      }

57      private static File getFile(File fparent, String rpath) {
58          File ret = new File(fparent, rpath);
59          if (ret.exists()) {
60              // OK;
61          } else {
62              ret = new File(fparent, rpath + ".xml");
63              if (! ret.exists()) {
64                  E.error("cant find file " + rpath + " in folder " + fparent + " (tried plain and adding .xml)");
65              }
66          }
67          return ret;
68      }

70  
71  
72      private static Object readFile(File f, XMLReader xmlr) {
73          Object ret = null;
74          if (f.exists()) {
75              String txt = FileUtil.readStringFromFile(f);
76              ret = xmlr.read(txt);
77          }
78          return ret;
79      }
