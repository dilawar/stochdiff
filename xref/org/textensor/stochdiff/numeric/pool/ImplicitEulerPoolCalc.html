9   /*
10   * This evolves concentrations such that the concentration change is computed from the
11   * concentrations AT THE END of the step. It uses a multidimensional Newton Raphson
12   * method to iterate to the solution to the concentration increments to be applied.
13   *
14   * The good thing about this is that it handles stiff equations nicely: you can tak a timestep that
15   *  is much bigger than the timescale of the fastest reaction. In particular, it handles an approach
16   *  to equilibrium nicely where some rates are so fast that the reactions are always in equilibrium.
17   *  You pay the price in accuracy if you use large timesteps, but that is not always so bad as it
18   *  seems as you may not care about the detailed time evolution of rapidly reacting species.
19   */

38      public double advance() {
39          double tol = 1.e-7;
40  
41          Column c = mconc;
42          Column dc = new Column(c.size());
43  
44  
45          int iit = 0;
46          double erhs = 0.;
47          do {
48              // fill rhs vector for step in conc dc, time dt;
49              Column rhs = rtab.stepResiduals(c, dc, dt);
50  
51              erhs =rhs.avgAbs();
52  
53              iit++;
54              if (erhs > tol) {
55                  if (iit > maxIterationsSoFar) {
56                      maxIterationsSoFar = iit;
57                  }
58  
59                  Matrix m = numderivs(c, dc, dt);
60                  Column w = m.LUSolve(rhs);
61                  dc.decrementBy(w);
62              }
63              if (iit > 12) {
64                  E.error(" failed to converge");
65                  break;
66              }
67          } while (erhs > 1.e-7);
68  
69          mconc.incrementBy(dc);
70  
71          return dt;
72      }

76      private final Matrix numderivs(Column vc, Column vdc,  double deltat) {
77          int n = rtab.nspecie;
78          Matrix m = new Matrix(n);
79  
80          double[] r0 = rtab.stepResiduals(vc, vdc, deltat).getData();
81          double delta = 1.e-6;
82  
83          for (int j = 0; j < n; j++) {
84              Column vdelta = vdc.copy();
85              vdelta.increment(j, delta);
86  
87              double[] rd = rtab.stepResiduals(vc, vdelta, deltat).getData();
88  
89              for (int i = 0; i < n; i++) {
90                  m.set(i, j, (rd[i] - r0[i]) / delta);
91              }
92          }
93          return m;
94      }
