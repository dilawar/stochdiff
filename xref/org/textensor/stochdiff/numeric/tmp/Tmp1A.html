191                 int irnp = (int) rnp;
192                 double drnp = rnp - irnp;
193 
194                 // random allocation to implement the remainder of the

196                 if (random.random() < drnp) {
197                     irnp += 1;
198                 }

200                 wkB[i][j] = irnp;
201             }

203             double a = surfaceAreas[i];
204             double[] scs = regsd[eltregions[i]];
205             if (a > 0 && scs != null) {
206                 for (int j = 0; j < nspec; j++) {
207                     if (Double.isNaN(scs[j])) {
208                         // means not specified by the user;
209 
210                     } else {
211                         wkA[i][j] = 0;
212                         wkB[i][j] = 0;
213 
214                         double rnp = a * scs[j] * PARTICLES_PUASD;
215                         int irnp = (int) rnp;
216                         double drnp = rnp - irnp;
217 
218                         // random allocation to implement the remainder of the
219                         // density (some cells get an extra particle, some dont)
220                         if (random.random() < drnp) {
221                             irnp += 1;
222                         }

224                         wkB[i][j] += irnp;

226                 }
227 
228             }

266             E.error("unknown probability distribution");
267         }
268 
269         if (doShared() || doParticle() || doIndependent()) {

271                 E.info("Using SHARED destination allocation");

273                 E.info("Using PER PARTICLE destination allocation");
274             }

276             pSharedOut = new double[nel][nspec];
277             fSharedExit = new double[nel][nspec][];
278 
279             int maxnn = 0;
280             for (int iel = 0; iel < nel; iel++) {
281                 for (int k = 0; k < nspec; k++) {
282                     int nn = neighbors[iel].length;
283                     fSharedExit[iel][k] = new double[nn];
284                     if (nn > maxnn) {
285                         maxnn = nn;
286                     }
287                 }
288             }
289             E.info("max no of neighbors for a single element is " + maxnn);
290 
291             for (int iel = 0; iel < nel; iel++) {
292                 for (int k = 0; k < nspec; k++) {
293                     int inbr[] = neighbors[iel];
294                     double lngnbr[] = lnCC[iel];
295                     int nnbr = inbr.length;
296                     // int np0 = wkA[iel][k];
297 
298                     double ptot = 0.;
299                     double[] pcnbr = new double[nnbr];
300 
301                     for (int j = 0; j < nnbr; j++) {
302                         double lnpgo = lnfdiff[k] + lngnbr[j] + lndt - lnvolumes[iel];
303                         // probability is dt * K_diff * contact_area /
304                         // (center_to_center_distance * source_volume)
305                         // gnbr contains the gometry: contact_area / distance
306 
307                         double p = Math.exp(lnpgo);
308                         ptot += p;
309                         pcnbr[j] = ptot;
310                     }
311 
312                     double lnptot = Math.log(ptot);
313                     if (lnptot > -1.) {
314                         // WK 9 11 2007
315                         System.out.println("WK===================================");
316                         System.out.println("In DIFFUSION: probability TOO HIGH!");
317                         System.out.println("Reduce your timestep, and try again...");
318                         System.out.println("WK====================================");
319                         System.exit(0);
320 
321                         /*
322                          * if (nwarn < 4) {
323                          * E.shortWarning("p too large at element " + iel +
324                          * " species " + k + " - capping from " +
325                          * Math.exp(lnptot) + " to " + Math.exp(-1.)); nwarn++;
326                          * } lnptot= -1.;
327                          */
328                         // WK
329                     }
330 
331                     pSharedOut[iel][k] = ptot;
332                     lnpSharedOut[iel][k] = lnptot;
333                     for (int j = 0; j < nnbr; j++) {
334                         fSharedExit[iel][k][j] = pcnbr[j] / ptot;
335                     }
336                 }
337             }
338         }

409     public final int run() {
410         init();
411 
412         if (resultWriter != null) {
413             resultWriter.writeString(vgrid.getAsText());
414             resultWriter.writeToSiblingFileAndClose(vgrid.getAsTableText(), "-mesh.txt");
415             for (int i = 0; i < fnmsOut.length; i++) {
416                 resultWriter.writeToSiblingFile(getGridConcsHeadings_dumb(i, vgrid), "-" + fnmsOut[i] + "-conc.txt");
417             }
418         }
419 
420         double time = 0.;
421         double runtime = sdRun.runtime;
422 
423         double tlog = 5.;
424 
425         long startTime = System.currentTimeMillis();
426 
427         // int iwr = 0;
428         double writeTime = -1.e-9;
429 
430         // RCC commenting this out to use desired output interval from model
431         // spec.
432         // sdRun.outputInterval = 100.0;
433         double[] writeTimeArray;
434         writeTimeArray = new double[fnmsOut.length];
435         for (int i = 0; i < fnmsOut.length; i++) {
436             writeTimeArray[i] = -1.e-9;
437             // System.out.println("writeTimeArray["+i+"] : " +
438             // writeTimeArray[i]);
439         }
440 
441         while (time < runtime) {
442 
443             if (time >= writeTime) {
444                 if (resultWriter != null) {
445                     resultWriter.writeString(getGridConcsText(time));
446                 }
447                 writeTime += sdRun.outputInterval;
448             }
449             for (int i = 0; i < fnmsOut.length; i++) {
450                 if (time >= writeTimeArray[i]) {
451                     resultWriter
452                     .writeToSiblingFile(getGridConcsPlainText_dumb(i, time), "-" + fnmsOut[i] + "-conc.txt");
453                     writeTimeArray[i] += Double.valueOf(dtsOut[i]);
454                 }
455             }
456 
457             time += advance(time);
458 
459             if (time > tlog) {
460                 E.info("time " + time + " dt=" + dt);
461                 tlog += Math.max(50 * sdRun.outputInterval, 5);
462             }
463 
464             if (time >= stateSaveTime) {
465                 resultWriter.writeToSiblingFile(getStateText(), sdRun.stateSavePrefix + Math.round(time) + ".nrds");
466                 stateSaveTime += sdRun.getStateSaveInterval();
467             }
468         }
469 
470         long endTime = System.currentTimeMillis();
471         E.info("total time " + (endTime - startTime) + "ms");
472 
473         return 0;
474     }

484     public double advance(double tnow) {
485 
486         // add in any injections
487         double[][] stims = stimTab.getStimsForInterval(tnow, dt);
488         for (int i = 0; i < stims.length; i++) {
489             double[] astim = stims[i];
490             for (int j = 0; j < astim.length; j++) {
491                 if (astim[j] > 0.) {
492                     // the stimulus could be spread over a number of elements
493                     // as yet, assume equal probability of entering any of these
494                     // elements (TODO)
495                     // the random < asr ensures we get the right number of
496                     // particles even the average entry per volume is less than
497                     // one
498                     // TODO - allow stim type (deterministic or poisson etc) in
499                     // config;
500 
501                     int nk = stimtargets[i].length;
502                     if (nk > 0) {
503                         double as = astim[j] / nk;
504                         int ias = (int) as;
505                         double asr = as - ((int) as);
506 
507                         for (int k = 0; k < nk; k++) {
508                             int nin = (ias + (random.random() < asr ? 1 : 0));
509                             ninjected += nin;
510 
511                             wkA[stimtargets[i][k]][j] += nin;
512                         }
513                     }
514                 }
515             }
516         }
517 
518         // initialize wkB to the current values.
519         // It will hold the midstep values for the leapfrog, after diffusion
520         // but before reactions.
521         for (int i = 0; i < nel; i++) {
522             for (int j = 0; j < nspec; j++) {
523                 wkB[i][j] = wkA[i][j];
524                 // if (wkB[i][j] < 0)
525                 // System.out.println("ERROR - NEGATIVE POPULATION at volume_element "
526                 // + i + ", specie " + j);
527                 // E.error("ERROR - NEGATIVE POPULATION at volume_element " + i
528                 // + ", specie " + j);
529             }
530         }
531 
532         // diffusion step;
533         for (int iel = 0; iel < nel; iel++) {
534 
535             for (int k = 0; k < nspec; k++) {
536                 if (lnfdiff[k] > -90) {
537 
538                     int np0 = wkA[iel][k];
539 
540                     if (np0 > 0) {
541                         switch (algoID) {
542                         case INDEPENDENT:
543                             // WK 8 28 2007
544                             // parallelDiffusionStep(iel, k);
545                             parallelAndSharedDiffusionStep(iel, k);
546                             // WK
547                             break;
548                         case SHARED:
549                             // WK 9 11 2007
550                             // sharedDiffusionStep(iel, k);
551                             parallelAndSharedDiffusionStep(iel, k);
552                             // WK
553                             break;
554                         case PARTICLE:
555                             particleDiffusionStep(iel, k);
556                             break;
557                         default:
558                             assert false;
559                         }
560                     }
561                 }
562             }
563         }
564 
565         // for the reaction step, the source array is wkB and the
566         // destination is wkA
567         for (int i = 0; i < nel; i++) {
568             for (int j = 0; j < nspec; j++) {
569                 wkA[i][j] = wkB[i][j];
570             }
571         }
572 
573         // reaction step;
574         for (int iel = 0; iel < nel; iel++) {
575             double lnvol = lnvolumes[iel];
576 
577             // start and end quantities for each species in a single
578             // volume
579             int[] nstart = wkB[iel];
580             int[] nend = wkA[iel];
581             for (int isp = 0; isp < nspecie; isp++) {
582                 nend[isp] = nstart[isp];
583             }
584 
585             for (int ireac = 0; ireac < nreaction; ireac++) {
586                 // total number of possible reactions is the number of
587                 // particles in the smallest reactant population
588 
589                 int[] ri = reactantIndices[ireac];
590                 int[] pi = productIndices[ireac];
591 
592                 int[] rs = reactantStochiometry[ireac];
593                 int[] ps = productStochiometry[ireac];
594 
595                 double lnp = lnrates[ireac] + lndt;
596 
597                 int n = nstart[ri[0]];
598 
599                 if (ri[1] >= 0) {
600                     int nk = nstart[ri[1]];
601 
602                     if (nk < n) {
603                         lnp += intlog(n);
604                         n = nk;
605                     } else {
606                         lnp += intlog(nk);
607                     }
608                     lnp -= lnvol;
609                     lnp -= LN_PARTICLES_PUVC;
610                 }
611 
612                 if (lnp > -1.) {
613                     if (nwarn < 5) {
614                         E.shortWarning("p too large at element " + iel + " reaction " + ireac + " capping from "
615                                        + Math.exp(lnp) + " to " + " exp(-1.)");
616                         nwarn++;
617                     }
618                     lnp = -1.;
619                 }
620 
621                 if (n <= 0) {
622 
623                 } else {
624                     int ngo = 0;
625                     if (n == 1) {
626                         // TODO use table to get rid of exp
627                         ngo = (random.random() < Math.exp(lnp) ? 1 : 0);
628                     } else if (n < StepGenerator.NMAX_STOCHASTIC) {
629                         ngo = interpSG.nGo(n, lnp, random.random());
630 
631                     } else {
632                         if (useBinomial()) {
633                             if (n * (Math.exp(lnp)) < NP) {
634                                 ngo = StepGenerator.gaussianStep(n, Math.exp(lnp), random.gaussian(), random.random(),
635                                                                  random.poisson(n * (Math.exp(lnp))), NP);
636                                 if (ngo < 0) {
637                                     ngo = 0;
638 
639                                     System.out.println("in advance (reaction), if (n*Math.exp(lnp)) < " + NP
640                                                        + "): ngo is NEGATIVE.");
641                                     System.out.println("ngo: " + ngo + " n: " + n + " Math.exp(lnp): " + Math.exp(lnp));
642                                 }
643                             } else {
644                                 ngo = StepGenerator.gaussianStep(n, Math.exp(lnp), random.gaussian(), random.random());
645                                 if (ngo < 0) {
646                                     ngo = 0;
647 
648                                     System.out.println("in advance (reaction), if (n*Math.exp(lnp)) >= " + NP
649                                                        + "): ngo is NEGATIVE.");
650                                     System.out.println("ngo: " + ngo + " n: " + n + " Math.exp(lnp): " + Math.exp(lnp));
651                                 }
652                             }
653 
654                         } else {
655                             ngo = StepGenerator.poissonStep(n, Math.exp(lnp), random.gaussian(), random.random());
656                             if (ngo < 0) {
657                                 ngo = 0;
658 
659                                 System.out.println("in advance (reaction), if not using binomial: ngo is NEGATIVE.");
660                                 System.out.println("ngo: " + ngo + " n: " + n + " Math.exp(lnp): " + Math.exp(lnp));
661                             }
662                         }
663                     }
664                     // WK 7 2 2008: if ngo is negative, exit.
665                     // if (ngo < 0)
666                     // {
667                     // System.out.println("in advance: ngo is NEGATIVE. Exiting...");
668                     // System.exit(0);
669                     // }
670                     // WK
671 
672                     // update the new quantities in npn;
673                     int ri0 = ri[0];
674                     int ri1 = ri[1];
675                     int rs0 = rs[0];
676                     int rs1 = rs[1];
677 
678                     int navail = nend[ri0] / rs[0];
679                     // AB changed navail > nend[ri1] / rs1 to navail < nend[ri1]
680                     // / rs1
681                     if (ri1 >= 0 && navail > nend[ri1] / rs1) {
682 
683                         navail = nend[ri1] / rs1;
684                     }
685                     if (ngo > navail) {
686                         // TODO as for diffusion, we've got more particles going
687                         // than there actually are. Should regenerate all
