5   /*
6    * Caching store of generators for cases where only a fixed number
7    * of different probabilities occur in the model. This applies to
8    * diffusion on a grid where all the elements have the same (or one
9    * of a few) volume and coupling constants. Then you get a different p
10   * for each diffusible species and for each different coupling constant.
11   *
12   * The way to use this is to create a DiscretePStepGenerator at the
13   * beginning, and then call getGenerator for each connection in the
14   * model. Then during the calculation, call generator.nGo(n, r) to
15   * get the number of particles that will move in one step.
16   *
17   *  Generators are cached in the hash map according to an integer
18   *  key based on their probability. Effectively, this splits the
19   *  space in log(p) between -20 and 0 into about 2E9 separate bins
20   *  and treats all p's in a single bin as the same. It avoids any
21   *  risk of creating separate generators for probabilities that
22   *  only differ by rounding errors.
23   *
24   *
25   */
26  
27  import org.textensor.stochdiff.numeric.BaseCalc.distribution_t;
28  
29  public class DiscretePStepGenerator extends StepGenerator {
30  
31  
32      HashMap<Integer, FixedPStepGenerator> generators;
33  
34  
35      public final static double lnpmin = -20;
36  
37      distribution_t mode;
38  
39      public DiscretePStepGenerator(distribution_t mode) {
40          generators = new HashMap<Integer, FixedPStepGenerator>();
41          this.mode = mode;
42      }
43  
44  
45      public FixedPStepGenerator getGenerator(double lnp) {
46          FixedPStepGenerator ret = null;
47          double po = lnp - lnpmin;
48          if (po < 0.) {
49              po = 0.;
50          }
51          int ipbin = (int)(Integer.MAX_VALUE * (po / (0. - lnpmin)));
52          Integer key = new Integer(ipbin);
53          if (generators.containsKey(key)) {
54              ret = generators.get(key);
55          } else {
56              ret = new FixedPStepGenerator(lnp, mode);
57              generators.put(key, ret);
58          }
59          return ret;
60      }
61  
62  
63      // this is here for completeness, but shouldnt be used
64      // in real calculations - you should call getGenerator once
65      // at the beginning and then use that - it will be much faster.
66      public int nGo(int n, double lnp, double r) {
67          return getGenerator(lnp).nGo(n, r);
68      }
