82      public final void init() {
83  
84          stateSaveTime = sdRun.getStateSaveInterval();
85          if (stateSaveTime <= 0.0) {
86              stateSaveTime = 1.e9;
87          }
88  
89  
90          rtab = getReactionTable();
91          vgrid = getVolumeGrid();
92  
93          nel = vgrid.getNElements();
94  
95          nspec = rtab.getNSpecies();
96          specieIDs = rtab.getSpecieIDs();
97  
98  
99  
100         volumes = vgrid.getElementVolumes();
101 
102         fdiff = rtab.getDiffusionConstants();
103         neighbors = vgrid.getPerElementNeighbors();
104         couplingConstants = vgrid.getPerElementCouplingConstants();
105 
106         extractOutputScheme(rtab); // see BaseCalc.java
107 
108         submembranes = vgrid.getSubmembranes();
109         regionLabels = vgrid.getRegionLabels();
110         // RO
111 
112         stimTab = getStimulationTable();
113         stimtargets = vgrid.getAreaIndexes(stimTab.getTargetIDs());
114         // eltstims gives the index in the stim array for
115         // the stim to element i, if any. -1 otherwise
116         /* AB Dec 16 2011 - eltstims needs to be 2D, determine target element and share for each */
117         int numStim=stimTab.getNStim();
118         eltstims = new int[numStim][nel];
119         eltstimshare = new double[numStim][nel];
120         for (int stimnum = 0; stimnum < numStim; stimnum++) {
121             for (int i = 0; i < eltstims[stimnum].length; i++) {
122                 eltstims[stimnum][i] = -1;
123             }
124         }
125         for (int i = 0; i < stimtargets.length; i++) {
126             //asti are the list of voxels to receive particles
127             int[] asti = stimtargets[i];
128             double vtot = 0.;
129             for (int k = 0; k < asti.length; k++) {
130                 vtot += volumes[asti[k]];
131             }
132 //AB 12-19-11: changed volumes[i] to volumes[asti[k]]
133             for (int k = 0; k < asti.length; k++) {
134                 eltstims[i][asti[k]] = i;
135                 eltstimshare[i][asti[k]] = volumes[asti[k]] / vtot;
136             }
137         }
138 
139         eltregions = vgrid.getRegionIndexes();
140         surfaceAreas = vgrid.getExposedAreas();
141 //AB 2012-apr 3 change wkB to wk[time-1]
142         wkA = new double[nel][nspec];
143         wktm1 = new double[nel][nspec];
144         wkC = new double[nel][nspec];
145 
146         dt = sdRun.fixedStepDt;
147 
148         double[][] regcon = getRegionConcentrations();
149         double[][] regsd = getRegionSurfaceDensities();
150 
151         for (int i = 0; i < nel; i++) {
152             double[] rcs = regcon[eltregions[i]];
153             for (int j = 0; j < nspec; j++) {
154                 wkA[i][j] = rcs[j];
155                 //AB 2012-apr 3 change wkB to wk[time-1]
156                 wktm1[i][j] = rcs[j];
157                 wkC[i][j] = rcs[j];
158             }
159 
160             double a = surfaceAreas[i];
161             double[] scs = regsd[eltregions[i]];
162             if (a > 0 && scs != null) {
163 
164                 // the actual concentration in the surface elements depends how
165                 // deep they are
166                 // scs in in nanomoles per square micron (check) surfaceArea in
167                 // square micron, so need to multiply
168                 // by surfaceArea to get nanomoles and divide by volume for
169                 // actual concentration
170                 double concfac = a / volumes[i];
171 
172                 for (int j = 0; j < nspec; j++) {
173                     if (Double.isNaN(scs[j])) {
174                         // means not specified by the user;
175 
176                     } else {
177                         wkA[i][j] = concfac * scs[j];
178                         //AB 2012-apr 3 change wkB to wk[time-1]
179                         wktm1[i][j] = concfac * scs[j];
180                         wkC[i][j] = concfac * scs[j];
181                     }
182                 }
183             }
184         }
185 
186         if (sdRun.initialStateFile != null) {
187             double[][] cc = readInitialState(sdRun.initialStateFile, nel, nspec, specieIDs);
188             if (cc != null) {
189                 for (int i = 0; i < nel; i++) {
190                     for (int j = 0; j < nspec; j++) {
191                         double c = cc[i][j];
192                         wkA[i][j] = c;
193                         //AB 2012-apr 3 change wkB to wk[time-1]
194                         wktm1[i][j] = c;
195                     }
196                 }
197             }
198         }
199

384     public double advance(double time) {
385         // diffusion terms;
386         // wkA is time t, wktm1 time t-1, wkC the next step, t+1
387 
388         // initialize next step values to zero;
389         for (int i = 0; i < nel; i++) {
390             for (int j = 0; j < nspec; j++) {
391                 wkC[i][j] = 0.;
392             }
393         }
394         // source terms;
395         // should increment injectiosn sites by 2 * the injected quantity;
396 
397         double[] zl = new double[nspec];
398         double[] zr = new double[nspec];
399 
400         // diffusion step;
401         for (int iel = 0; iel < nel; iel++) {
402             double vol = volumes[iel];
403             double fvol = 2. * dt / vol; // 2 for the two leapfrog steps;
404 
405             // zr part not needed unless we put the source terms in wkC first
406             for (int k = 0; k < nspec; k++) {
407                 zr[k] = wkC[iel][k] / vol;
408                 zl[k] = 1.;
409             }
410 
411             int inbr[] = neighbors[iel];
412             double gnbr[] = couplingConstants[iel];
413             int nnbr = inbr.length;
414             /*
415             			for (int j = 0; j < nnbr; j++) {
416             				for (int k = 0; k < nspec; k++) {
417             					double ff = fvol * fdiff[k] * gnbr[j];
418             					//AB 2012-apr 3 change wkB to wk[time-1], and wkA to wk[time]
419             					zr[k] += ff * (wkA[inbr[j]][k] - 0.5 * wktm1[iel][k]);
420             					zl[k] += 0.5 * ff;
421             				}
422             			}
423             			for (int k = 0; k < nspec; k++) {
424             				wkC[iel][k] = (wkA[iel][k] + zr[k]) / zl[k];
425             			}
426             */
427             //AB 2012 Apr 4 - skip the diffusion if fdiff=0 to make this part faster
428             for (int k = 0; k < nspec; k++) {
429                 if (fdiff[k]>0) {
430                     for (int j = 0; j < nnbr; j++) {
431                         double ff = fvol * fdiff[k] * gnbr[j];
432                         zr[k] += ff * (wkA[inbr[j]][k] - 0.5 * wktm1[iel][k]);
433                         zl[k] += 0.5 * ff;
434                     }
435                 }
436                 wkC[iel][k] = (wkA[iel][k] + zr[k]) / zl[k];
437             }
438         }
439 
440         double[][] stims = stimTab.getStimsForInterval(time, dt);
441 
442         // reaction step;
443         for (int iel = 0; iel < nel; iel++) {
444 //AB 12-19-11	create the coninc outside of the pinj loop, to accumulate injections from multiple stimuli
445             double[] concinc=new double[nspec];
446             int concnull=1;
447             for (int k=0; k<nspec; k++) {
448                 concinc[k]=0.0;
449             }
450             double fconc = NM_PER_PARTICLE_PUV / volumes[iel];
451 //AB 12-19-11 need to create loop over all the stimulations, and
452 //			create pinj for each stimulation (not just the first one)
453             for (int stimnum=0; stimnum<stims.length; stimnum++) {
454                 if (eltstims[stimnum][iel] >= 0) {
455                     concnull=0;
456                     double[] pinj=stims[stimnum];
457 
458                     for (int i = 0; i < pinj.length; i++) {
459                         concinc[i] += pinj[i] * fconc * eltstimshare[stimnum][iel];
460                         if (concinc[i] < 0) {
461                             E.error("negative concentration? " + concinc[i]);
462                         }
463                     }
464                 }
465             }
466             if (concnull==0) {
467                 reacStep(wkC[iel], dt, concinc);
468             } else {
469                 reacStep(wkC[iel], dt, null);
470             }
471         }
472 
473         // cycle the solution arrays
474         //AB 2012-apr 3 change wkB to wk[time-1]
475         double[][] wkT = wktm1;
476         wktm1 = wkA;
477         wkA = wkC;
478         wkC = wkT;
479 
480         return dt;
481     }

483     private void reacStep(double[] concs, double deltat, double[] concinc) {
484         // Semi-implicit euler - see pool/SemiImplicitEulerPoolCalc
485         Column col = new Column(concs);
486         Matrix m = rtab.getIncrementRateMatrix(col);
487         Column cp = rtab.getProductionColumn(col);
488         Column cpdt = cp.times(deltat);
489 
490         /*
491          * if (nlog < 5) { E.info("concs " + concs[0] + " " + concs[1] + " " +
492          * concs[2]); nlog++; m.dump(); }
493          */
494 
495         m.multiplyBy(dt);
496         m.subtractIdentity();
497         m.negate();
498 
499         /*
500          * if (nlog < 5) { m.dump(); }
501          */
502 
503         if (concinc != null) {
504             cpdt.incrementBy(concinc);
505         }
506 
507         Column dc = m.LUSolve(cpdt);
508         col.incrementBy(dc);
509         col.writeTo(concs); // TODO noop?
510     }
