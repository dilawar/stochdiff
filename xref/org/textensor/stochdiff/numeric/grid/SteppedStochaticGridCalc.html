42  /*
43   * Approximate stochastic calculation with a fixed timestep where
44   * the number of reactions orruring in a volume and the number of particles
45   * moving to neighbouring volumes are both generated from lookup tables
46   * in p (probability of event for one particle)  and
47   * n (number of particles of a given type)  for a uniform random number.
48   *
49   * This is approximate because the effect of the particles on thier new
50   * location is note taken into account until the following step.
51   *
52   *
53   * Units: concentrations are expressed in nM and volumes in cubic microns
54   * So, in these units , one Litre is 10^15 and a 1M solution is 10^9
55   * The conversion factor between concentrations and particle number is
56   * therefore
57   * nparticles = 6.022^23 * vol/10^15 * conc/10^9
58   * ie, nparticles = 0.6022 * conc
59   *
60   *
61   */

157     public final void init() {
158 
159         stateSaveTime = sdRun.getStateSaveInterval();
160         if (stateSaveTime <= 0.0) {
161             stateSaveTime = 1.e9;
162         }
163         stateSaveTime += sdRun.getStartTime();
164 
165         // something to generate the random nunmbers
166         random = new MersenneTwister(getCalculationSeed());
167 
168         rtab = getReactionTable();
169         speciesIDs = rtab.getSpeciesIDs();
170 
171         nreaction = rtab.getNReaction();
172         rates = rtab.getRates();
173 
174         // Debug.dump("rates", rates);
175 
176         lnrates = ArrayUtil.log(rates, -999.);
177 
178         reactantIndices = rtab.getReactantIndices();
179         productIndices = rtab.getProductIndices();
180 
181         reactantStochiometry = rtab.getReactantStochiometry();
182         productStochiometry = rtab.getProductStochiometry();
183 
184         vgrid = getVolumeGrid();
185 
186         nel = vgrid.getNElements();
187 
188         nspec = rtab.getNSpecies();
189         specieIDs = rtab.getSpecieIDs();
190         volumes = vgrid.getElementVolumes();
191         lnvolumes = ArrayUtil.log(volumes, -999.);
192 
193         // RO
194         // ----------------------
195         // System.out.println("Number of files        : " + NspeciesFilef);
196         // System.out.println("Total numer of species : " + NspeciesIDsOutf);
197 
198         // ----------------------
199         // RO
200         extractOutputScheme(rtab); // see BaseCalc.java
201 
202         surfaceAreas = vgrid.getExposedAreas();
203 
204         // WK 6 18 2007
205         submembranes = vgrid.getSubmembranes();
206         regionLabels = vgrid.getRegionLabels();
207         // WK
208 
209         fdiff = rtab.getDiffusionConstants();
210         lnfdiff = ArrayUtil.log(fdiff, -999.);
211 
212         neighbors = vgrid.getPerElementNeighbors();
213         couplingConstants = vgrid.getPerElementCouplingConstants();
214         lnCC = ArrayUtil.log(couplingConstants, -999.);
215 
216         stimTab = getStimulationTable();
217         stimtargets = vgrid.getAreaIndexes(stimTab.getTargetIDs());
218 
219         // workspace for the calculation
220         wkA = new int[nel][nspec];
221         wkB = new int[nel][nspec];
222         wkReac = new int[nreaction];
223 
224         eltregions = vgrid.getRegionIndexes();
225         double[][] regcon = getRegionConcentrations();
226         double[][] regsd = getRegionSurfaceDensities();
227 
228         // apply initial conditions over the grid
229         for (int i = 0; i < nel; i++) {
230             double v = volumes[i];
231             double[] rcs = regcon[eltregions[i]];
232 
233             for (int j = 0; j < nspec; j++) {
234                 double rnp = v * rcs[j] * PARTICLES_PUVC;
235                 int irnp = (int) rnp;
236                 double drnp = rnp - irnp;
237 
238                 // random allocation to implement the remainder of the
239                 // density (some cells get an extra particle, some dont)
240                 if (random.random() < drnp) {
241                     irnp += 1;
242                 }
243                 wkA[i][j] = irnp;
244                 wkB[i][j] = irnp;
245             }
246 
247             double a = surfaceAreas[i];
248             double[] scs = regsd[eltregions[i]];
249             if (a > 0 && scs != null) {
250                 for (int j = 0; j < nspec; j++) {
251                     if (Double.isNaN(scs[j])) {
252                         // means not specified by the user;
253 
254                     } else {
255                         wkA[i][j] = 0;
256                         wkB[i][j] = 0;
257 
258                         double rnp = a * scs[j] * PARTICLES_PUASD;
259                         int irnp = (int) rnp;
260                         double drnp = rnp - irnp;
261 
262                         // random allocation to implement the remainder of the
263                         // density (some cells get an extra particle, some dont)
264                         if (random.random() < drnp) {
265                             irnp += 1;
266                         }
267                         wkA[i][j] += irnp;
268                         wkB[i][j] += irnp;
269                     }
270                 }
271 
272             }
273 
274             /*
275              * if (i % 20 == 0) { E.info("elt " + i + " region " + eltregions[i]
276              * + " n0 " + wkA[i][0]); }
277              */
278         }
279 
280         if (sdRun.initialStateFile != null) {
281             double[][] cc = readInitialState(sdRun.initialStateFile, nel, nspec, speciesIDs);
282             if (cc != null) {
283                 for (int i = 0; i < nel; i++) {
284                     for (int j = 0; j < nspec; j++) {
285                         int np = (int) Math.round(cc[i][j] * volumes[i] / CONC_OF_N);
286                         wkA[i][j] = np;
287                         wkB[i][j] = np;
288                     }
289                 }
290             }
291         }
292 
293         dt = sdRun.fixedStepDt;
294         lndt = Math.log(dt);
295 
296         // take logs of integers once only and store;
297         intlogs = new double[10000];
298         intlogs[0] = -99;
299         for (int i = 1; i < intlogs.length; i++) {
300             intlogs[i] = Math.log(i);
301         }
302 
303         // final things we need is something to generate particle numbers
304         // for steps of given n, p
305         if (useBinomial())
306             interpSG = InterpolatingStepGenerator.getBinomialGenerator();
307         else if (usePoisson()) {
308             interpSG = InterpolatingStepGenerator.getPoissonGenerator();
309         } else {
310             E.error("unknown probability distribution");
311         }
312 
313         if (doShared() || doParticle() || doIndependent()) {
314             if (doShared()) {
315                 E.info("Using SHARED destination allocation");
316             } else {
317                 E.info("Using PER PARTICLE destination allocation");
318             }
319             lnpSharedOut = new double[nel][nspec];
320             pSharedOut = new double[nel][nspec];
321             fSharedExit = new double[nel][nspec][];
322 
323             int maxnn = 0;
324             for (int iel = 0; iel < nel; iel++) {
325                 for (int k = 0; k < nspec; k++) {
326                     int nn = neighbors[iel].length;
327                     fSharedExit[iel][k] = new double[nn];
328                     if (nn > maxnn) {
329                         maxnn = nn;
330                     }
331                 }
332             }
333             E.info("max no of neighbors for a single element is " + maxnn);
334 
335             for (int iel = 0; iel < nel; iel++) {
336                 for (int k = 0; k < nspec; k++) {
337                     int inbr[] = neighbors[iel];
338                     double lngnbr[] = lnCC[iel];
339                     int nnbr = inbr.length;
340                     // int np0 = wkA[iel][k];
341 
342                     double ptot = 0.;
343                     double[] pcnbr = new double[nnbr];
344 
345                     for (int j = 0; j < nnbr; j++) {
346                         double lnpgo = lnfdiff[k] + lngnbr[j] + lndt - lnvolumes[iel];
347                         // probability is dt * K_diff * contact_area /
348                         // (center_to_center_distance * source_volume)
349                         // gnbr contains the gometry: contact_area / distance
350 
351                         double p = Math.exp(lnpgo);
352                         ptot += p;
353                         pcnbr[j] = ptot;
354                     }
355 
356                     double lnptot = Math.log(ptot);
357                     if (lnptot > -1.) {
358                         // WK 9 11 2007
359                         System.out.println("WK===================================");
360                         System.out.println("In DIFFUSION: probability TOO HIGH!");
361                         System.out.println("Reduce your timestep, and try again...");
362                         System.out.println("WK====================================");
363                         System.exit(0);
364 
365                         /*
366                          * if (nwarn < 4) {
367                          * E.shortWarning("p too large at element " + iel +
368                          * " species " + k + " - capping from " +
369                          * Math.exp(lnptot) + " to " + Math.exp(-1.)); nwarn++;
370                          * } lnptot= -1.;
371                          */
372                         // WK
373                     }
374 
375                     pSharedOut[iel][k] = ptot;
376                     lnpSharedOut[iel][k] = lnptot;
377                     for (int j = 0; j < nnbr; j++) {
378                         fSharedExit[iel][k][j] = pcnbr[j] / ptot;
379                     }
380                 }
381             }
382         }
383     }

386     private String getGridConcsText(double time) {
387         StringBuffer sb = new StringBuffer();
388         // TODO tag specific to integer quantities;
389         int nspecout = ispecout.length;
390         if (nspecout == 0) {
391             return "";
392         }
393 
394         sb.append("gridConcentrations " + nel + " " + nspecout + " " + time + " ");
395         for (int i = 0; i < nspecout; i++) {
396             sb.append(specieIDs[ispecout[i]] + " ");
397         }
398         sb.append("\n");
399 
400         for (int i = 0; i < nel; i++) {
401             for (int j = 0; j < nspecout; j++) {
402                 if (writeConcentration) {
403                     sb.append(stringd((CONC_OF_N * wkA[i][ispecout[j]] / volumes[i])));
404 
405                 } else {
406                     sb.append(stringi(wkA[i][ispecout[j]]));
407                 }
408             }
409             sb.append("\n");
410         }
411         return sb.toString();
412     }

455     public final int run() {
456         init();
457 
458         if (resultWriter != null) {
459             if (!resultWriter.isContinuation()) {
460                 resultWriter.writeString(vgrid.getAsText());
461             }
462             if (vgrid.isCuboid()) {
463                 resultWriter.writeToSiblingFileAndClose(vgrid.getAsTableText(), "-mesh.txt");
464 
465             } else if (vgrid.isCurved()) {
466                 resultWriter.writeToSiblingFileAndClose(vgrid.getAsTableText(), "-mesh.txt");
467 
468                 resultWriter.writeToSiblingFileAndClose(vgrid.getAsElementsText(), "-elements.tri");
469                 E.info("Written elements mesh file");
470 
471                 File ftri = resultWriter.getSiblingFile("-elements.tri");
472 
473                 if (runAction == VISUALIZE) {
474                     CCViz3D sdv = new CCViz3D();
475                     sdv.loadElements(ftri);
476                     E.info("loaded mesh in standalone viewer");
477                     sdv.show();
478                     return 1;
479                 }
480             } else {
481                 E.error("Unrecognized grid - neither cuboid or curved");
482             }
483 
484             for (int i = 0; i < fnmsOut.length; i++) {
485                 String sibsuf = "-" + fnmsOut[i] + "-conc.txt";
486                 String shead = getGridConcsHeadings_dumb(i, vgrid);
487                 StringTokenizer st = new StringTokenizer(shead);
488                 int nt = st.countTokens();
489 
490                 if (resultWriter.isContinuation()) {
491                     ResultWriter sibrw = resultWriter.getRawSibling(ResultWriter.TEXT, sibsuf, null);
492                     sibrw.pruneFrom("", 0, sdRun.getStartTime());
493                     sibrw.init(null, ResultWriter.TEXT);
494                 } else {
495                     resultWriter.writeToSiblingFile(shead, sibsuf);
496                 }
497             }
498         }
499 
500         double time = sdRun.getStartTime();
501         double endtime = sdRun.getEndTime();
502 
503         E.info("Running from time=" + time + "ms to time=" + endtime + "ms");
504 
505         double tlog = 5.;
506 
507         long startTime = System.currentTimeMillis();
508 
509         // int iwr = 0;
510         double writeTime = time - 1.e-9;
511 
512         double[] writeTimeArray;
513         writeTimeArray = new double[fnmsOut.length];
514         for (int i = 0; i < fnmsOut.length; i++) {
515             writeTimeArray[i] = -1.e-9;
516             // System.out.println("writeTimeArray["+i+"] : " +
517             // writeTimeArray[i]);
518         }
519 
520         while (time < endtime) {
521 
522             if (time >= writeTime) {
523                 if (resultWriter != null) {
524                     resultWriter.writeString(getGridConcsText(time));
525                 }
526                 writeTime += sdRun.outputInterval;
527             }
528             for (int i = 0; i < fnmsOut.length; i++) {
529                 if (time >= writeTimeArray[i]) {
530                     resultWriter.writeToSiblingFile(getGridConcsPlainText_dumb(i, time), "-" + fnmsOut[i] + "-conc.txt");
531                     writeTimeArray[i] += Double.valueOf(dtsOut[i]);
532                 }
533             }
534 
535 
536             time += advance(time);
537 
538             if (time > tlog) {
539                 E.info("time " + time + " dt=" + dt);
540                 tlog += Math.max(50 * sdRun.outputInterval, 5);
541             }
542 
543             if (time >= stateSaveTime) {
544                 resultWriter.writeToFinalSiblingFile(getStateText(), sdRun.stateSavePrefix + Math.round(time) + ".nrds");
545                 stateSaveTime += sdRun.getStateSaveInterval();
546             }
547         }
548 
549         long endTime = System.currentTimeMillis();
550         E.info("total time " + (endTime - startTime) + "ms");
551         return 0;
552

563     public double advance(double tnow) {
564 
565         // add in any injections
566         double[][] stims = stimTab.getStimsForInterval(tnow, dt);
567         for (int i = 0; i < stims.length; i++) {
568             double[] astim = stims[i];
569             for (int j = 0; j < astim.length; j++) {
570                 if (astim[j] > 0.) {
571                     // the stimulus could be spread over a number of elements
572                     // as yet, assume equal probability of entering any of these
573                     // elements (TODO)
574                     // the random < asr ensures we get the right number of
575                     // particles even the average entry per volume is less than
576                     // one
577                     // TODO - allow stim type (deterministic or poisson etc) in
578                     // config;
579 
580                     int nk = stimtargets[i].length;
581                     if (nk > 0) {
582                         double as = astim[j] / nk;
583                         int ias = (int) as;
584                         double asr = as - ((int) as);
585 
586                         for (int k = 0; k < nk; k++) {
587                             int nin = (ias + (random.random() < asr ? 1 : 0));
588                             ninjected += nin;
589 
590                             wkA[stimtargets[i][k]][j] += nin;
591                         }
592                     }
593                 }
594             }
595         }
596 
597         // initialize wkB to the current values.
598         // It will hold the midstep values for the leapfrog, after diffusion
599         // but before reactions.
600         for (int i = 0; i < nel; i++) {
601             for (int j = 0; j < nspec; j++) {
602                 wkB[i][j] = wkA[i][j];
603                 // if (wkB[i][j] < 0)
604                 // System.out.println("ERROR - NEGATIVE POPULATION at volume_element "
605                 // + i + ", specie " + j);
606                 // E.error("ERROR - NEGATIVE POPULATION at volume_element " + i
607                 // + ", specie " + j);
608             }
609         }
610 
611         // diffusion step;
612         for (int iel = 0; iel < nel; iel++) {
613 
614             for (int k = 0; k < nspec; k++) {
615                 if (lnfdiff[k] > -90) {
616 
617                     int np0 = wkA[iel][k];
618 
619                     if (np0 > 0) {
620 
621                         switch(algoID) {
622                         case INDEPENDENT:
623                             // WK 8 28 2007
624                             // parallelDiffusionStep(iel, k);
625                             parallelAndSharedDiffusionStep(iel, k);
626                             // WK
627                             break;
628                         case SHARED:
629                             // WK 9 11 2007
630                             // sharedDiffusionStep(iel, k);
631                             parallelAndSharedDiffusionStep(iel, k);
632                             // WK
633                             break;
634                         case PARTICLE:
635                             particleDiffusionStep(iel, k);
636                             break;
637 
638                         default:
639                             assert false;
640                         }
641                     }
642                 }
643             }
644         }
645 
646         // for the reaction step, the source array is wkB and the
647         // destination is wkA
648         for (int i = 0; i < nel; i++) {
649             for (int j = 0; j < nspec; j++) {
650                 wkA[i][j] = wkB[i][j];
651             }
652         }
653 
654         // reaction step;
655         for (int iel = 0; iel < nel; iel++) {
656             double lnvol = lnvolumes[iel];
657 
658             // start and end quantities for each species in a single
659             // volume
660             int[] nstart = wkB[iel];
661             int[] nend = wkA[iel];
662             for (int isp = 0; isp < nspecie; isp++) {
663                 nend[isp] = nstart[isp];
664             }
665 
666             for (int ireac = 0; ireac < nreaction; ireac++) {
667                 // total number of possible reactions is the number of
668                 // particles in the smallest reactant population
669 
670                 int[] ri = reactantIndices[ireac];
671                 int[] pi = productIndices[ireac];
672 
673                 int[] rs = reactantStochiometry[ireac];
