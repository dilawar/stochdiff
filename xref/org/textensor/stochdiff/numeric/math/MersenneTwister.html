59      public void setSeed(final long seed) {
60          mt = new int[N];
61  
62          mag01 = new int[2];
63          mag01[0] = 0x0;
64          mag01[1] = MATRIX_A;
65  
66          mt[0] = (int)(seed & 0xffffffff);
67          for (mti = 1; mti < N; mti++) {
68              mt[mti] = (1812433253 * (mt[mti - 1] ^ (mt[mti - 1] >>> 30)) + mti);
69              /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
70              /* In the previous versions, MSBs of the seed affect */
71              /* only MSBs of the array mt[]. */
72              /* 2002/01/09 modified by Makoto Matsumoto */
73              mt[mti] &= 0xffffffff;
74              /* for >32 bit machines */
75          }
76      }

80      public final float random() {
81          int y;
82  
83          // generate a block of words for use later;
84          if (mti >= N) {
85              int kk;
86  
87              // local copies are faster
88              final int[] mtl = mt;
89              final int[] mag01l = mag01;
90  
91              for (kk = 0; kk < N - M; kk++) {
92                  y = (mtl[kk] & UPPER_MASK) | (mtl[kk + 1] & LOWER_MASK);
93                  mtl[kk] = mtl[kk + M] ^ (y >>> 1) ^ mag01l[y & 0x1];
94              }
95              for (; kk < N - 1; kk++) {
96                  y = (mtl[kk] & UPPER_MASK) | (mtl[kk + 1] & LOWER_MASK);
97                  mtl[kk] = mtl[kk + (M - N)] ^ (y >>> 1) ^ mag01l[y & 0x1];
98              }
99              y = (mtl[N - 1] & UPPER_MASK) | (mtl[0] & LOWER_MASK);
100             mtl[N - 1] = mtl[M - 1] ^ (y >>> 1) ^ mag01l[y & 0x1];
101 
102             mti = 0;
103         }
104 
105         y = mt[mti++];
106         y ^= y >>> 11; // TEMPERING_SHIFT_U(y)
107         y ^= (y << 7) & TEMPERING_MASK_B; // TEMPERING_SHIFT_S(y)
108         y ^= (y << 15) & TEMPERING_MASK_C; // TEMPERING_SHIFT_T(y)
109         y ^= (y >>> 18); // TEMPERING_SHIFT_L(y)
110 
111         return (y >>> 8) / ((float)(1 << 24));
112     }

118     public final double gaussian() {
119         double ret = 0.;
120         if (haveGaussian) {
121             ret = spareGaussian;
122             haveGaussian = false;
123         } else {
124             double r = -1;
125             double ran1 = 0;
126             double ran2 = 0;
127             while (r <= 0.0 || r >= 1.0) {
128                 ran1 = 2 * random() - 1;
129                 ran2 = 2 * random() - 1;
130                 r = ran1 * ran1 + ran2 * ran2;
131             }
132             double fac = Math.sqrt(-2. * Math.log(r) / r);
133             ret = ran1 * fac;
134             spareGaussian = ran2 * fac;
135             haveGaussian = true;
136         }
137         return ret;
138     }

148     public final double gammln(double xx) {
149         double x = xx - 1.0;
150         double tmp = x + 5.5;
151         tmp -= (x+0.5) * Math.log(tmp);
152         double ser = 1.0;
153         for (int j = 0; j <= 5; j++) {
154             x += 1.0;
155             ser += cof[j]/x;
156         }
157         return -tmp+Math.log(2.50662827465*ser);
158     }

162     public final int poisson(double mean) {
163         // In "Numerical Recipes" Ch 7-3 p.294
164         double em = 0.;
165         if (mean < 12.0) {
166             double g=Math.exp(-mean);
167             em= -1;
168             double t=1.0;
169             do {
170                 ++em;
171                 t *= random();
172 
173             } while (t > g);
174 
175         } else {
176             double sq = Math.sqrt(2.0*mean);
177             double alxm=Math.log(mean);
178             double g = mean*alxm - gammln(mean+1.0);
179             double t = 0.;
180             double y = 0.;
181             do {
182                 do {
183                     y = Math.tan(Math.PI * random());
184                     em = sq*y + mean;
185                 }  while (em < 0.0);
186 
187                 em = Math.floor(em);
188                 t = 0.9*(1.0 + y*y) * Math.exp(em*alxm - gammln(em + 1.0) -g);
189 
190             } while (random() > t);
191         }
192 
193         int ret = (int)(Math.round(em));
194         return ret;
195     }
